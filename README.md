# casos-de-dengue-hospital
datos de el hospital por entradas salidas y relacion que tiene estos cassos con hospitales de bogota
PROYECTO BIG DATA HOSPITAL - MONGODB + NODE.JS // ============================================================= // INSTALAR DEPENDENCIAS: // npm init -y // npm install express mongoose cors dotenv // EJECUTAR: // node server.js // ======================================================== // server.js // ======================================================= require('dotenv').config(); const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const app = express(); // ======================================================= // CONFIGURACIONES // ====================================================== app.use(cors()); app.use(express.json()); const PUERTO = proceso.env.PUERTO || 3000; // ======================================================= // CONEXIÓN A MONGODB // ============================================================= mongoose.connect('mongodb://127.0.0.1:27017/hospital_bigdata') .then(() => { console.log('MongoDB conectado correctamente'); }) .catch((error) => { console.log('Error de conexión:', error); }); // ============================================================== // MODELO PACIENTE // ======================================================== const pacienteSchema = new mongoose.Schema({ id_paciente: { tipo: Número, requerido: verdadero }, departamento_reporte: { tipo: Cadena, requerido: verdadero }, municipio_reporte: { tipo: Cadena, requerido: verdadero }, codigo_evento: Número, ano_reporte: Número, mes_reporte: Número, semana_epidemiologica: Número, edad: Número, uni_med: Cadena, nacionalidad: Cadena, sexo: Cadena, regimen_de_salud: Cadena, pertenencia_etnica: String, nom_upgd: String, nombre_del_evento: String, pais_reporte: String }); const Paciente = mongoose.model('Paciente', pacienteSchema); // ============================================================== // RUTAS CRUD // =============================================================== // ------------------------------------------------------ // OBTENER TODOS LOS PACIENTES // ------------------------------------------------------ app.get('/api/pacientes', async (req, res) => { try { const pacientes = await Paciente.find(); res.json(pacientes); } catch (error) { res.status(500).json({ mensaje: error.message }); } }); // ------------------------------------------------------ // REGISTRAR PACIENTE // ------------------------------------------------------ app.post('/api/pacientes', async (req,res) => { try { const nuevoPaciente = new Paciente({ id_paciente: req.body.id_paciente, departamento_reporte: req.body.departamento_reporte, municipio_reporte: req.body.municipio_reporte, codigo_evento: req.body.codigo_evento, ano_reporte: req.body.ano_reporte, mes_reporte: req.body.mes_reporte, semana_epidemiologica: req.body.semana_epidemiologica, edad: req.body.edad, uni_med: req.body.uni_med, nacionalidad: req.body.nacionalidad, sexo: req.body.sexo, regimen_de_salud: req.body.regimen_de_salud, pertenencia_etnica: req.body.pertenencia_etnica, nom_upgd: req.body.nom_upgd, nombre_del_evento: req.body.nombre_del_evento, pais_reporte: req.body.pais_reporte }); const pacienteGuardado = await nuevoPaciente.save(); res.status(201).json({ mensaje: 'Paciente registrado correctamente', pacienteGuardado }); } catch (error) { res.status(400).json({ mensaje: error.message }); } }); // ------------------------------------------------------ // ACTUALIZAR PACIENTE // ------------------------------------------------------ app.put('/api/pacientes/:id', async (req, res) => { try { const pacienteActualizado = await Paciente.findByIdAndUpdate( req.params.id, req.body, { new: true } ); res.json({ mensaje: 'Paciente actualizado', pacienteActualizado }); } catch (error) { res.status(500).json({ mensaje: error.message }); } }); // ------------------------------------------------------ // ELIMINAR PACIENTE // ------------------------------------------------------ app.delete('/api/pacientes/:id', async (req, res) => { try { await Paciente.findByIdAndDelete(req.params.id); res.json({ mensaje: 'Paciente eliminado correctamente' }); } catch (error) { res.status(500).json({ mensaje: error.message }); } }); // ====================================================== // CONSULTAS BIG DATA // ====================================================== // ------------------------------------------------------ // TOTAL PACIENTES // ------------------------------------------------------ app.get('/api/estadisticas/total', async (req, res) => { try { const total = await Paciente.countDocuments(); res.json({ total_pacientes: total }); } catch (error) { res.status(500).json({ mensaje: error.message }); } }); // ------------------------------------------------------ // PROMEDIO DE EDADES // ------------------------------------------------------ app.get('/api/estadisticas/promedio-edad', async (req, res) => { try { const promedio = await Paciente.aggregate([ { $group: { _id: null, promedio_edad: { $avg: '$edad' } } } ]); res.json(promedio); } catch (error) { res.status(500).json({ mensaje: error.message }); } }); // ------------------------------------------------------ // PACIENTES POR SEXO // ------------------------------------------------------ app.get('/api/estadisticas/sexo', async (req, res) => { try { const resultado = await Paciente.aggregate([ { $group: { _id: '$sexo', cantidad: { $sum: 1 } } } ]); res.json(resultado); } catch (error) { res.status(500).json({ mensaje: error.message }); } }); // ====================================================== // INSERTAR DATOS DE PRUEBA // ====================================================== app.get('/api/datos-prueba', async (req, res) => { try { const paciente = new Paciente({ id_paciente: 1001, departamento_reporte: 'Bogotá', municipio_reporte: 'Suba', codigo_evento: 210, ano_reporte: 2026, mes_reporte: 5, semana_epidemiologica: 18, edad: 25, uni_med: 'Hospital Central', nacionalidad: 'Colombiana', sexo: 'Masculino', regimen_de_salud: 'Contributivo', pertenencia_etnica: 'Sin pertenencia étnica', nom_upgd: 'UPGD Central', nombre_del_evento: 'Dengue', pais_reporte: 'Colombia' }); await paciente.save(); res.json({ mensaje: 'Datos de prueba insertados' }); } catch (error) { res.status(500).json({ mensaje: error.message }); } }); // ====================================================== // SERVIDOR // ====================================================== app.listen(PORT, () => { console.log(`Servidor ejecutándose en puerto ${PORT}`); }); // ====================================================== // FIN DEL PROYECTO // ====================================================== ```log(`Servidor ejecutándose en puerto ${PORT}`); }); // ====================================================== // FIN DEL PROYECTO // ====================================================== ```
